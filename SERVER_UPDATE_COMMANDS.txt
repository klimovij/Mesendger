Команды для обновления на сервере:

cd ~/mesendger/telegram-clone

# 1. Принудительно обновить код из GitHub
git fetch origin --all
git checkout main
git reset --hard origin/master || git reset --hard origin/main

# 2. Пересобрать клиент
cd client-react
npm install
npm run build
